81@app.route("/users/<username>/bookings", methods=['GET']) 

82def user_bookings(username): 

83 """ 

84 Gets booking information from the 'Bookings Service' for the user, and 

85 movie ratings etc. from the 'Movie Service' and returns a list. 

86 :param username: 

87 :return: List of Users bookings 

88 """ 

89 if username not in users: 

90 raise NotFound("User '{}' not found.".format(username)) 

91 

92 try: 

93 users_bookings = requests.get("http://{}:{}/bookings/{}".format(helper.resolve_requests_host('bookings'), helper.get_port('bookings'), username), timeout=helper.get_timeout('bookings')) 

94 except requests.exceptions.ConnectionError: 

95 raise ServiceUnavailable("The Bookings service is unavailable.") 

96 except requests.exceptions.Timeout: 

97 raise ServiceUnavailable("The Bookings service timed out.") 

98 

99 if users_bookings.status_code == 404: 

100 raise NotFound("No bookings were found for {}".format(username)) 

101 

102 if users_bookings.status_code != 200: 

103 raise ServiceUnavailable("The Bookings service is malfunctioning.") 

104 

105 users_bookings = users_bookings.json() 

106 

107 # For each booking, get the rating and the movie title 

108 result = {} 

109 for date, movies in users_bookings.items(): 

110 result[date] = [] 

111 for movieid in movies: 

112 try: 

113 movies_resp = requests.get("http://{}:{}/movies/{}".format(helper.resolve_requests_host('movies'), helper.get_port('movies'), movieid), timeout=helper.get_timeout('movies')) 

114 except requests.exceptions.ConnectionError: 

115 raise ServiceUnavailable("The Movie service is unavailable.") 

116 except requests.exceptions.Timeout: 

117 raise ServiceUnavailable("The Movie service timed out.") 

118 

119 if movies_resp.status_code != 200: 

120 raise ServiceUnavailable("The Movie service is malfunctioning.") 

121 

122 movies_resp = movies_resp.json() 

123 

124 result[date].append({ 

125 "title": movies_resp["title"], 

126 "rating": movies_resp["rating"], 

127 "uri": movies_resp["uri"] 

128 })
